DATA SEGMENT
	PORTA EQU 00H
	PORTB EQU 04H
	PORTC EQU 08H
	PORT_CON EQU 10H
DATA ENDS

CODE SEGMENT PUBLIC 'CODE'
	ASSUME CS: CODE

MOV AX, DATA
MOV DS, AX
ORG 0000H

MOV DX, PORT_CON
MOV AL, 82H
OUT DX, AL

START: 
	MOV DX, PORTB
	IN AL, DX
	
	CMP AL, 11111110B
	JE ZERO
	CMP AL, 11111101B
	JE ONE
	CMP AL, 11111011B
	JE TWO
	CMP AL, 11110111B
	JE THREE
	CMP AL, 11101111B 
	JE FOUR
	CMP AL, 11011111B
	JE FIVE
	CMP AL, 10111111B	
	JE SIX
	CMP AL, 01111111B
	JE SEVEN
                      ;3FH, 06H, 5BH, 4FH, 66H, 6DH, 7DH, 07H

ZERO : MOV AL, 3FH
       MOV DX, PORTA
       OUT DX, AL
       JMP START

ONE : MOV AL, 06H
       MOV DX, PORTA
       OUT DX, AL
       JMP START

TWO: MOV AL, 5BH
     MOV DX, PORTA
     OUT DX, AL
     JMP START

THREE : MOV AL, 4FH
       MOV DX, PORTA
       OUT DX, AL
       JMP START

FOUR : MOV AL, 66H
       MOV DX, PORTA
       OUT DX, AL
       JMP START

FIVE : MOV AL, 6DH
       MOV DX, PORTA
       OUT DX, AL
       JMP START

SIX:   MOV AL, 7DH
       MOV DX, PORTA
       OUT DX, AL
       JMP START

SEVEN : MOV AL, 07H
       MOV DX, PORTA
       OUT DX, AL
       JMP START

CODE ENDS
	END START
